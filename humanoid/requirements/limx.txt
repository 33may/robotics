# Humanoid repo Python dependencies — `limx` conda env (Python 3.8).
#
# Purpose: runtime sandbox for any process that imports `limxsdk` —
#   - humanoid-mujoco-sim/simulator.py
#   - humanoid-rl-deploy-python/main.py + controllers
#   - python3 -m limxsdk.ability.cli ...
#   - any future bridge that publishes/subscribes on the LimX MROS bus
#
# Hard ABI lock: the limxsdk wheel embeds libpython3.8.so.1.0 and
# limxsdk/robot/_robot.so is built against the Python 3.8 C-API.
# This env is Python 3.8 and stays Python 3.8.
#
# Out of scope: training, Isaac Sim, LeRobot, anything torch.
#
# Install:
#   conda create -n limx -c conda-forge python=3.8.18 -y
#   conda activate limx
#   pip install vendor/humanoid-mujoco-sim/limxsdk-lowlevel/python3/amd64/limxsdk-4.0.1-py3-none-any.whl
#   pip install -r requirements/limx.txt
#
# Version pins below are the last PyPI releases that still ship cp38
# manylinux wheels (verified 2026-06-19). Do not bump without verifying
# cp38 wheel availability AND compatibility with the SDK's declared
# constraints (mujoco>3.2.2, numpy<1.26.4,>1.21.0).

# Core sim + inference
mujoco==3.2.3            # last cp38 wheel; SDK requires >3.2.2 (exact intersection)
onnxruntime==1.19.2      # last cp38 wheel — walk/mimic controllers
numpy==1.24.4            # last cp38 wheel; SDK requires <1.26.4,>1.21.0
scipy==1.10.1            # last cp38 wheel — Rotation in walk/mimic

# SDK-declared transitive deps (kept explicit so we control versions)
pyyaml>=6,<7
pygame                   # used by SDK joystick layer
pandas                   # used by SDK diagnostics

# Undeclared SDK runtime deps — used by limxsdk but NOT listed in the wheel METADATA
click                    # limxsdk/ability/cli.py imports click at module top; missing it = ModuleNotFoundError on `cli load`/`cli switch`
